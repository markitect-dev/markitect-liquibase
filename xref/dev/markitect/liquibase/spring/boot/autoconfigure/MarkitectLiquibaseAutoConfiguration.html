52  @AutoConfiguration(
53      before = LiquibaseAutoConfiguration.class,
54      after = {DataSourceAutoConfiguration.class, HibernateJpaAutoConfiguration.class})
55  @ConditionalOnClass({SpringLiquibase.class, DatabaseChange.class})
56  @ConditionalOnProperty(prefix = "spring.liquibase", name = "enabled", matchIfMissing = true)
57  @Conditional(LiquibaseDataSourceCondition.class)
58  @Import(DatabaseInitializationDependencyConfigurer.class)
59  public class MarkitectLiquibaseAutoConfiguration {
60    @Bean
61    public SpringLiquibaseBeanPostProcessor springLiquibaseBeanPostProcessor(
62        Environment environment) {
63      return new SpringLiquibaseBeanPostProcessor(environment);
64    }
65  
66    @Configuration(proxyBeanMethods = false)
67    @ConditionalOnClass(ConnectionCallback.class)
68    @ConditionalOnMissingBean(SpringLiquibase.class)
69    @EnableConfigurationProperties({LiquibaseProperties.class, MarkitectLiquibaseProperties.class})
70    public static class MarkitectLiquibaseConfiguration {
71      private final LiquibaseProperties properties;
72      private final MarkitectLiquibaseProperties markitectProperties;
73  
74      MarkitectLiquibaseConfiguration(
75          LiquibaseProperties properties, MarkitectLiquibaseProperties markitectProperties) {
76        Assert.notNull(properties, "Properties must not be null");
77        Assert.notNull(markitectProperties, "MarkitectProperties must not be null");
78        this.properties = properties;
79        this.markitectProperties = markitectProperties;
80      }
81  
82      @Bean
83      SpringLiquibase liquibase(
84          @SuppressWarnings("SpringJavaInjectionPointsAutowiringInspection") @LiquibaseDataSource
85              ObjectProvider<DataSource> liquibaseDataSource,
86          ObjectProvider<DataSource> dataSource) {
87        return toSpringLiquibase(liquibaseDataSource.getIfAvailable(), dataSource.getIfUnique());
88      }
89  
90      private SpringLiquibase toSpringLiquibase(
91          @Nullable DataSource liquibaseDataSource, @Nullable DataSource dataSource) {
92        DataSource migrationDataSource = toMigrationDataSource(liquibaseDataSource, dataSource);
93        MarkitectSpringLiquibase liquibase = new MarkitectSpringLiquibase();
94        liquibase.setDropFirst(properties.isDropFirst());
95        liquibase.setClearCheckSums(properties.isClearChecksums());
96        liquibase.setShouldRun(properties.isEnabled());
97        liquibase.setDataSource(migrationDataSource);
98        liquibase.setChangeLog(properties.getChangeLog());
99        liquibase.setContexts(properties.getContexts());
100       liquibase.setLabelFilter(properties.getLabels());
101       liquibase.setTag(properties.getTag());
102       liquibase.setDefaultSchema(properties.getDefaultSchema());
103       liquibase.setLiquibaseTablespace(properties.getLiquibaseTablespace());
104       liquibase.setLiquibaseSchema(properties.getLiquibaseSchema());
105       liquibase.setDatabaseChangeLogTable(properties.getDatabaseChangeLogTable());
106       liquibase.setDatabaseChangeLogLockTable(properties.getDatabaseChangeLogLockTable());
107       liquibase.setTestRollbackOnUpdate(properties.isTestRollbackOnUpdate());
108       liquibase.setChangeLogParameters(properties.getParameters());
109       liquibase.setRollbackFile(properties.getRollbackFile());
110       liquibase.setCloseDataSourceOnceMigrated(
111           migrationDataSource != liquibaseDataSource && migrationDataSource != dataSource);
112       liquibase.setOutputDefaultCatalog(markitectProperties.isOutputDefaultCatalog());
113       liquibase.setOutputDefaultSchema(markitectProperties.isOutputDefaultSchema());
114       return liquibase;
115     }
116 
117     private DataSource toMigrationDataSource(
118         @Nullable DataSource liquibaseDataSource, @Nullable DataSource dataSource) {
119       if (liquibaseDataSource != null) {
120         return liquibaseDataSource;
121       }
122       String url = properties.getUrl();
123       if (url != null) {
124         DataSourceBuilder<?> builder = DataSourceBuilder.create();
125         builder.type(SimpleDriverDataSource.class);
126         builder.url(url);
127         applyCommonBuilderProperties(builder);
128         return builder.build();
129       }
130       Assert.state(dataSource != null, "Liquibase migration DataSource missing");
131       if (properties.getUser() != null) {
132         DataSourceBuilder<?> builder = DataSourceBuilder.derivedFrom(dataSource);
133         builder.type(SimpleDriverDataSource.class);
134         applyCommonBuilderProperties(builder);
135         return builder.build();
136       }
137       return dataSource;
138     }
139 
140     private void applyCommonBuilderProperties(DataSourceBuilder<?> builder) {
141       Optional.ofNullable(properties.getDriverClassName())
142           .filter(StringUtils::hasText)
143           .ifPresent(builder::driverClassName);
144       builder.username(properties.getUser());
145       builder.password(properties.getPassword());
146     }
147   }

152     }

161   }

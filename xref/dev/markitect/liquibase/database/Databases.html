17  package dev.markitect.liquibase.database;
18  
19  import static dev.markitect.liquibase.structure.Structures.isCatalogOrSchemaType;
20  import static dev.markitect.liquibase.util.Preconditions.checkNotNull;
21  import static dev.markitect.liquibase.util.Strings.isIllegalIdentifier;
22  import static liquibase.util.BooleanUtil.isTrue;
23  
24  import java.util.Locale;
25  import java.util.function.BiPredicate;
26  import liquibase.GlobalConfiguration;
27  import liquibase.database.AbstractJdbcDatabase;
28  import liquibase.database.ObjectQuotingStrategy;
29  import liquibase.structure.DatabaseObject;
30  import org.checkerframework.checker.nullness.qual.Nullable;
31  
32  public final class Databases {
33    public static <D extends AbstractJdbcDatabase> @Nullable String escapeObjectName(
34        D database,
35        BiPredicate<String, Class<? extends DatabaseObject>> mustQuoteObjectName,
36        @Nullable String objectName,
37        Class<? extends DatabaseObject> objectType) {
38      checkNotNull(database);
39      checkNotNull(mustQuoteObjectName);
40      checkNotNull(objectType);
41      if (objectName == null) {
42        return null;
43      }
44      return mustQuoteObjectName.test(objectName, objectType)
45          ? database.quoteObject(database.correctObjectName(objectName, objectType), objectType)
46          : objectName;
47    }
48  
49    public static <D extends AbstractJdbcDatabase> boolean mustQuoteObjectName(
50        D database,
51        @Nullable Boolean unquotedObjectsAreUppercased,
52        String objectName,
53        Class<? extends DatabaseObject> objectType) {
54      checkNotNull(database);
55      checkNotNull(objectName);
56      checkNotNull(objectType);
57      return database.getObjectQuotingStrategy() == ObjectQuotingStrategy.QUOTE_ALL_OBJECTS
58          || isIllegalIdentifier(objectName)
59          || database.isReservedWord(objectName)
60          || (unquotedObjectsAreUppercased != null
61              && isCatalogOrSchemaType(objectType)
62              && isTrue(GlobalConfiguration.PRESERVE_SCHEMA_CASE.getCurrentValue())
63              && !objectName.equals(
64                  unquotedObjectsAreUppercased
65                      ? objectName.toUpperCase(Locale.US)
66                      : objectName.toLowerCase(Locale.US)));
67    }
68  
69    public static <D extends AbstractJdbcDatabase> @Nullable String correctObjectName(
70        D database,
71        @Nullable Boolean unquotedObjectsAreUppercased,
72        @Nullable String objectName,
73        Class<? extends DatabaseObject> objectType) {
74      checkNotNull(database);
75      checkNotNull(objectType);
76      if (objectName == null) {
77        return null;
78      }
79      if (unquotedObjectsAreUppercased == null
80          || database.getObjectQuotingStrategy() == ObjectQuotingStrategy.QUOTE_ALL_OBJECTS
81          || (isCatalogOrSchemaType(objectType)
82              && isTrue(GlobalConfiguration.PRESERVE_SCHEMA_CASE.getCurrentValue()))) {
83        return objectName;
84      }
85      if (unquotedObjectsAreUppercased) {
86        return objectName.toUpperCase(Locale.US);
87      }
88      return objectName.toLowerCase(Locale.US);
89    }
90  
91    private Databases() {}
92  }
